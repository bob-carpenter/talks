\sld{The problem}
\begin{itemize}
\item Given a Bayesian \myemph{posterior density} $p(\theta \mid y),$
  with support for \myemph{parameters} $\theta \in \mathbb{R}^D$ and \myemph{data} $y$,
  draw a \myemph{sample}
  $$
  \draw{\theta}{1}, \ldots \draw{\theta}{M} \sim p(\theta \mid y)
  $$
\item to evaluate \myemph{posterior expectations} of functions $f$
\begin{align}
  \expect{f(\theta) \mid y}
  &= \int_{\mathbb{R}^D} f(\theta) \cdot p(\theta \mid y) \,
    \textrm{d}\theta
  \\[4pt]
  &= \lim_{M \rightarrow \infty} \frac{1}{M} \sum_{m=1}^M 
f\left( \draw{\theta}{m} \right)
  \\[4pt]
  &\approx \frac{1}{M} \sum_{m=1}^M f\left(\draw{\theta}{m}\right) 
\end{align}
\end{itemize}

\sld{Bayesian quantities of interest}
\begin{itemize}
\item \myemph{Parameter estimate} minimizing expected square error
  $$\widehat{\theta} = \expect{\theta \mid y}$$
\item \myemph{Event probability} for event $A \subseteq \mathbb{R}^D$
  $$\Pr[A \mid y] = \expect{\textrm{I}(\theta \in A) \mid y}$$
\item \myemph{Posterior predictive density} for new data $\widetilde{y}$
  $$p(\widetilde{y} \mid y) = \expect{p(\widetilde{y} \mid \theta)
    \mid y}.$$
\end{itemize}

\sld{What's the problem?}
\begin{itemize}
\item Some target densities $p(\theta \mid y)$ are very hard to sample
\item The major reasons why they are hard:
  \begin{subitemize}
  \item very \myemph{high dimensional} or expensive to evaluate
  \item difficult to compute gradients
  \item very \myemph{badly conditioned}
  \item {\huge $\star$} multiple target scales (i.e., varying curvature)
  \end{subitemize}
\end{itemize}
